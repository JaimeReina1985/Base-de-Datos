# Importar bibliotecas necesarias
import pandas as pd
from transformers import BertTokenizer
import torch

import warnings

warnings.simplefilter(action='ignore', category=UserWarning)
pd.options.mode.chained_assignment = None  # default='warn'

# Cargar el conjunto de datos
file_path = 'Noticias.xlsx'
data = pd.read_excel(file_path)

# Eliminar las filas con valores faltantes
data.dropna(inplace=True)

# Preprocesar los datos
tokenizer = BertTokenizer.from_pretrained('bert-base-uncased')
data['input_ids'] = data['contenido'].apply(lambda x: tokenizer.encode(x, add_special_tokens=True))
max_len = 128
data['input_ids'] = data['input_ids'].apply(lambda x: x[:max_len] + [0]*(max_len-len(x)) if len(x) < max_len else x[:max_len])

## Etiqueta para detectar deportes y filtrar noticias de archivo
data=data[data['Etiqueta']!="archivo"]
data['Etiqueta'] = [1 if x == "deportes" else 0 for x in data["Etiqueta"]]

# Dividir el conjunto de datos en entrenamiento y validacin
from sklearn.model_selection import train_test_split
X_train, X_val, y_train, y_val = train_test_split(data['input_ids'].tolist(), data['Etiqueta'].tolist(), test_size=0.2, random_state=42)

# Conversin a tensores de PyTorch
train_encodings = torch.tensor(X_train)
val_encodings = torch.tensor(X_val)

# Convertir etiquetas a tensores y ajustar su forma a (num_samples, 1).
# Se cambia reshape
train_labels = torch.tensor(y_train, dtype=torch.float32).unsqueeze(1)
val_labels = torch.tensor(y_val, dtype=torch.float32).unsqueeze(1)

## Creacin de mscaras de atencin
# Genera mscaras de atencin (1.0 para tokens reales, 0.0 para padding)
train_masks = torch.tensor([[float(i != 0) for i in ii] for ii in X_train])
val_masks = torch.tensor([[float(i != 0) for i in ii] for ii in X_val])

Documentacion

  • Se realiza carga de librerias y base de datos.
  • Se realiza limpieza de los datos, eliminando filas con valores faltantes
  • Se usa la tokenizacion de BERT para convertir el texto de cada noticia en una secuencia de numeros.
  • Luego, todas las secuencias se recortan o rellenan (padding) para que tengan la misma longitud (128 en este caso). Es importante para no trabajar con secuencias de diferentes tamaos
  • Se eliminan las noticias cuya etiqueta sea "archivo", ya que solo nos interesan las categorias especializadas.
  • Despues, cambiamos la columna de etiquetas para que sea Binaria 1 ponga 1 si es "Deportes" y 0 para las demas. Asi convertimos el problema en una clasificacion binaria.
  • Separamos los datos en dos partes: entrenamiento (80%) y validacion (20%), para poder entrenar el modelo.
  • Convertimos las listas tokens y etiquetas a tensores de PyTorch, ya que BERT espera recibir los datos en este formato.
  • Por ultimo, creamos las "attention mask", que son arreglos que indican al modelo que posiciones del input son palabras reales y cuales son solo relleno (padding). Esto ayuda a BERT para que ignore el padding durante el entrenamiento.

from sklearn.metrics import f1_score, roc_auc_score, accuracy_score
from transformers import EvalPrediction
import torch
import numpy as np  # Asegrate de importar numpy

# Desactiva WANDB para evitar logging externo.
import os
os.environ["WANDB_DISABLED"] = "true"

# fuente: https://jesusleal.io/2021/04/21/Longformer-multilabel-classification/
def multi_label_metrics(predictions, labels, threshold=0.5):

    # Aplica la funcin sigmoide a las predicciones (logits) para obtener probabilidades.
    sigmoid = torch.nn.Sigmoid()
    probs = sigmoid(torch.Tensor(predictions))

    # Convierte las probabilidades en predicciones binarias (0 o 1) usando el umbral.
    y_pred = np.zeros(probs.shape)
    y_pred[np.where(probs >= threshold)] = 1

    # Calcula las mtricas F1-score (micro), ROC AUC (micro) y Accuracy.
    y_true = labels
    f1_micro_average = f1_score(y_true=y_true, y_pred=y_pred, average='micro')
    roc_auc = roc_auc_score(y_true, y_pred, average = 'micro')
    accuracy = accuracy_score(y_true, y_pred)

    # devuelve las mtricas como un diccionario
    metrics = {'f1': f1_micro_average,
               'roc_auc': roc_auc,
               'accuracy': accuracy}
    return metrics

def compute_metrics(p: EvalPrediction):
    # Obtiene las predicciones (logits) del modelo
    preds = p.predictions[0] if isinstance(p.predictions, tuple) else p.predictions

    # Llama a la funcin 'multi_label_metrics' para calcular las mtricas.
    result = multi_label_metrics(
        predictions=preds,
        labels=p.label_ids)
    return result

Documentacion

  • f1_score, roc_auc_score, accuracy_score de sklearn.metrics: Sirven para calcular mtricas muy comunes en tareas de clasificacin.
  • EvalPrediction de transformers: Se usa para facilitar el manejo de las predicciones y etiquetas en el entrenamiento/evaluacin de modelos con Hugging Face.
  • torch: Usado para trabajar con tensores y funciones como la sigmoide.
  • Se crea un diccionario "metrics", con los resultados.
